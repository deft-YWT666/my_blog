### 并查集

```cpp
void init_set() //初始化并查集 , 每个节点的'老大'都是自己.
{
    for(int i = 1 ; i <= n ; i++) s[i] = i ;
}

int find_set(int x) //不带路径压缩
{
    if(x == s[x]) return x ;
    else
        return find_set(s[x]) ;
}

int find_set(int x) //路径压缩 , 提高查找'老大'的效率
{
    if(x != s[x]) s[x] = find_set(s[x]) ;
    return s[x] ;
}

void merge_set(int x , int y) // 合并
{
    x = find_set(x) ;
    y = find_set(y) ;
    if(x != y) s[x] = s[y] ;
}
```



### 快速幂

```cpp
#define int long long
int ksm(int a , int n , int mod) // 计算 a的n次幂 再对mod取模
{
    int ans = 1 ;
    a %= mod ;
    while(n)
    {
        if(n & 1) ans = (ans * a) % mod ;
        // n 的二进制与 000...001 做'与'运算,即看最后一位是0还是1
        
        a = (a * a) % mod ;
        n >>= 1 ;
        //n 进位后, 下一位对应权值为 a^2 
    }
    return ans ;
}
```

```cpp
void solve()
{
  int n , m ; // 以n个顶点 , m条边的图为例
  cin >> n >> m ;
  memset(e , 0x3f , sizeof(e)) ;// memset中第二个参数0x3f表示把每条边初始化为无穷大
  for(int i = 1 ; i <= m ; i ++)
  {
    int u , v , w ;
    cin >> u >> v >> w ;
    e[u][v] = min(e[u][v] , w) ;
    e[v][u] = e[u][v] ; //无向图不解释
  }
  for(int k = 1 ; k <= n ; k++)
    for(int i = 1 ; i <= n ; i++)
      for(int j = 1 ; j <= n ; j++)
        e[i][j] = min(e[i][j] , e[i][k] + e[k][j]) ;
}
```

### 判断素数

```cpp
bool is_prime(int x){
    if(x <= 1) return false;
    // 1 不是素数
    for(int i = 2; i <= x / i; i++){
        if(x % i == 0) return false;
    }
    return true;
    //注意 2 , 3 都是素数 上面那个循环需要 x >= 4时才会运行 
}


void init(){
	is_prime[0] = is_prime[1] = false;
	for(int i = 2; i <= n; i++){
		if(is_prime[i]){
			for(int j = i * i; j <= n; j += i){
				is_prime[j] = false;
			}
		}
	}
}
//素数筛
```

#### LCM 和 GCD

```cpp
// 计算两个数的最大公约数
int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}

// 计算两个数的最小公倍数（避免溢出）
int lcm(int a, int b) {
    return a / gcd(a, b) * b;  // 先除后乘，避免溢出
}
```
